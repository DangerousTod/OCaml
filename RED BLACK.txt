type color = 
    Red
   | Black
  type 'a rbtree =
    Node of color * 'a * 'a rbtree * 'a rbtree
   | Leaf
  let rec mem x = function
    Leaf -> false
   | Node (_, y, left, right) ->
      x = y || (x < y && mem x left) || (x > y && mem x right)
  ;;
